def check_validity
  if @shared_link[:url].nil?
    raise ArgumentError, "Missing `:url` option"
  end

  @shared_link.keys.each do |key|
    unless VALID_KEYS.include? key
      raise ArgumentError, "Invalid option `#{key.inspect}`"
    end
  end
end
