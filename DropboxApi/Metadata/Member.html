def initialize(member)
  @member_hash = case member
    when Hash
      member
    when String
      hash_from_email_or_dropbox_id member
    when DropboxApi::Metadata::Member
      member.to_hash
    else
      raise ArgumentError, "Invalid object for Member: #{member.inspect}"
    end
end
