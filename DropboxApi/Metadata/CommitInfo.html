def build_from_options(options)
  options = Hash[options.map do |key, value|
    case key
    when :mode
      [key.to_s, build_write_mode(value)]
    when :client_modified
      [key.to_s, build_client_modified(value)]
    when :path, :autorename, :mute
      [key.to_s, value]
    end
  end.compact]

  new(options)
end
